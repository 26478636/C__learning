命令行 ----- git基本操作

// ------------------------------------------
环境配置
git config --global user.name ""
git config --global user.email ""
查看当前的用户名和用户邮箱
git config user.name
git config user.email
git config --list // 查看更多的信息，q退出

// ------------------------------------------
初始化本地仓库
git init // 初始化带工作区的本地仓库
git init --bare // 初始化不带工作区的本地仓库

// ------------------------------------------
git clone

// ------------------------------------------
查看文件的状态
git status
git status -s // 简介版本

// ------------------------------------------
将文件加入到暂存区
git add <文件名>
将暂存区的文件取消
git reset <文件名>

// ------------------------------------------
将暂存区的文件，提交到本地仓库
git commit -m "" <文件名>

// ------------------------------------------
从本地仓库，删除文件
git rm <文件名>

// ------------------------------------------
