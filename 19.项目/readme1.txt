Linux：多用户(允许同时登录多个用户进行操作)，多任务(允许用户同时进行多个操作任务)；
Windows：单用户，多任务；

Linux一切皆文件

基本指令
pwd                 // 打印当前终端所在目录
// ---------------------------------------------------------------------

ls                  // 列出当前工作目录下的所有文件/文件夹的名称
ls ./               // 相对路径 表示当前目录下
ls ../              // 表示上一目录下
ls /home            // 绝对路径 
****ls + 选项 + 路径****
ls -lah /home       // -l list 表示以详细列表的形式展示
                    // -a 表示显示所有的文件/文件夹(包括隐藏的)
                    // -h 表示以可读性较高的形式显示 
// ---------------------------------------------------------------------

cd  或者  cd~       // 直接进入到当前用户的home目录下
cd ../             // 相对路径 进入到上级目录下
cd ../local        // 进入到上级目录中的local目录下
cd /usr/local      // 绝对路径 进入到/user/local目录下
// ---------------------------------------------------------------------

mkdir              // 创建目录
mkdir myfolder     // 在当前目录下，创建出目录"myfolder"
mkdir -p           // 一次性创建**多层**不存在的目录，别忘了-p
mkdir -p a/b/c 
mkdir a/ b/ c/     // 一次性创建多个目录
// ---------------------------------------------------------------------

touch              // 创建文件的
touch linux.txt    // 在当前目录下，创建linux.txt文件
touch ../linux.txt // 在上级目录下，创建linux.txt文件
touch /root/wm/C__learing/linux.txt // 绝对路径
touch linux linux.txt               // 在当前目录下，创建这两个文件
// ---------------------------------------------------------------------

rm                 // 删除文件/文件夹
rm linux.txt       // 删除当前目录下的linux.txt文件
rm /root/wm/C__learing/19.项目/linux.txt  // 绝对路径
rm -rf myfolder    // 删除当前目录下的myfolder文件夹
rm -rf /root/wm/C__learing/myfolder
// ---------------------------------------------------------------------

cp                  // 复制文件/文件夹到指定位置
cp /root/wm/C__learing/1.txt ./   // 复制1.txt(文件)到当前目录下
cp -r /root/wm/C__learing/1 ./    // 复制1(文件夹)到当前目录下
// ---------------------------------------------------------------------

mv                      // 移动文件到新的位置，或者重命名文件
mv 1.txt ../            // 将1.txt文件，移动到上一级目录下
mv 1 ../                // 将1文件夹，移动到上一级目录下
mv 1.txt ../123.txt     // 将1.txt文件，移动到上一级目录下，并命名为123.txt
// ---------------------------------------------------------------------

man                     // 查看命令使用手册，按q退出
man ls
man cd 
man man
...
// ---------------------------------------------------------------------

reboot                  // 重启Linux系统
// ---------------------------------------------------------------------

shutdown -h now         // 立即重启
